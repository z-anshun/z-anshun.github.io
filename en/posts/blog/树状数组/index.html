func reversePairs(nums []int) int {
	n := len(nums)
	tmp := make([]int, n)
	copy(tmp, nums)
	sort.Ints(tmp) // æ’åº

	for i := 0; i < n; i++ {
		// nums = {3,4,2}
		// å¦‚ï¼Œtmp={2,3,4}
		// nums = {2,3,1}
		nums[i] = sort.SearchInts(tmp, nums[i]) + 1 // æ‰¾åˆ°å¯¹åº”çš„ä¸‹æ ‡ +1
	}

	bit := BIT{n: n, tree: make([]int, n+1)}
	ans := 0
	// ä»åå¾€å‰éå†. ç°åœ¨numsä¸ºæ’åºåçš„å¯¹åº”çš„index
	for i := n - 1; i >= 0; i-- {
		ans += bit.query(nums[i] - 1)
		bit.update(nums[i])
	}
	return ans
}

type BIT struct {
	n    int
	tree []int
}

func (b BIT) lowbit(x int) int {
	return x & (-x)
}

func (b BIT) query(x int) int {
	ret := 0
	for x > 0 {
		ret += b.tree[x]
		x -= b.lowbit(x)
	}
	return ret
}

// å¯¹åº”çš„+1
// è¿™é‡Œå¯èƒ½ä¼šå¾ˆå¤§
func (b BIT) update(x int) {
	for x <= b.n {
		b.tree[x]++
		x += b.lowbit(x)
	}
}
