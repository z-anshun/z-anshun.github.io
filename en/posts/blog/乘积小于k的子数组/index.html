func numSubarrayProductLessThanK(nums []int, k int) int {
    res:=0
    n:=len(nums)
    l:=0
    now:=1
    for r:=0;r<n;r++{
        now*=nums[r]
        for l<r&&now>=k{
            now/=nums[l]
            l++
        }
        if r>=l&&now<k{
            res+=r-l+1
        }

    }
    return res
}
