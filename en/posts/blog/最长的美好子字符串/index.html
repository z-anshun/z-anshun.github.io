func longestNiceSubstring(s string) (ans string) {
	if s == "" {
		return
	}
	lower, upper := 0, 0
	for _, ch := range s {
		if unicode.IsLower(ch) {
			lower |= 1 << (ch - 'a')
		} else {
			upper |= 1 << (ch - 'A')
		}
	}
	// å¦‚æœå½“å‰å³ä¸ºå®Œç¾å­—ç¬¦ä¸²
	if lower == upper {

	}

func longestNiceSubstring(s string) (ans string) {
	mask := uint(0)
	for _, ch := range s {
		mask |= 1 << (unicode.ToLower(ch) - 'a')
	}

	maxNum := bits.OnesCount(mask)

	for i := 1; i <= maxNum; i++ {
		var lowerC, upperC [26]int
		total, cnt, l := 0, 0, 0
		for r, ch := range s {
			idx := unicode.ToLower(ch) - 'a'
			if ch >= 'a' && ch <= 'z' {
				lowerC[idx]++
				if lowerC[idx] == 1 && upperC[idx] > 0 {
					cnt++
				}
			} else {
				upperC[idx]++
				if upperC[idx] == 1 && lowerC[idx] > 0 {
					cnt++
				}
			}
			// åªå«æœ‰ä¸€ä¸ª
			if lowerC[idx]+upperC[idx] == 1 {
				total++
			}
			// å½“æ€»çš„ä¸ªæ•°å¤§äºå½“å‰ç±»å‹ä¸ªæ•°äº†
			for total > i {
				idx := unicode.ToLower(rune(s[l])) - 'a'
				if lowerC[idx]+upperC[idx] == 1 {
					total--
				}
				if s[l] >= 'a' && s[l] <= 'z' {
					lowerC[idx]--
					if lowerC[idx] == 0 && upperC[idx] > 0 {
						cnt--
					}
				} else {
					upperC[idx]--
					if upperC[idx] == 0 && lowerC[idx] > 0 {
						cnt++
					}
				}
				l++
			}
			if cnt == i && r-l+1 > len(ans) {
				ans = s[l : r+1]
			}
		}

	}
	return
}

func longestSubstring(s string, k int) int {

	var r [26]int
	for _, v := range s {
		r[v-'a']++
	}
	var split byte
	for i, v := range r[:] {
		if v > 0 && v < k {
			split = byte(i) + 'a'
			break
		}
	}

	if split == 0 {
		return len(s)
	}
	strs := strings.Split(s, string(split))
	res := 0
	for _, str := range strs {
		res = max(res, longestSubstring(str, k))
	}
	return res

}

func max(x, y int) int {
	if x > y {
		return x
	}
	return y
}
