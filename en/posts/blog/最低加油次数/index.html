func minRefuelStops(target int, startFuel int, stations [][]int) int {
	var qp []int
	res := 0

	nowTank := startFuel
	for _, station := range stations {
		if nowTank < station[0] {
			// é€‰å¤šä¸ªè·¯è¿‡çš„ç«™åŠ 
			if len(qp) == 0 {
				return -1
			}
			c:=0
			for r:=0;r<len(qp);r++{
				nowTank += qp[r]
				res++
				c++
				if nowTank>=station[0]{
					break
				}
			}
			qp=qp[c:]
		}
		if nowTank >= station[0] {
			qp = append(qp, station[1])
			sort.Slice(qp, func(i, j int) bool {
				return qp[i] > qp[j]
			})
		} else {
			return -1
		}
	}
	if nowTank >= target {
		return res
	}
	for i := 0; i < len(qp); i++ {
		nowTank += qp[i]
		res++
		if target <= nowTank {
			return res
		}
	}

	return -1

}
